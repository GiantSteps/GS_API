    [docs]	def getLastEvents(self,pattern,step,num,stepSize):
		events = []
		for i  in reversed(range(1,num+1)):
			idx = step - i*stepSize;
			if idx < 0 :
				idx+=pattern.duration
			events += [pattern.getStartingEventsAtTime(idx)]
		return events
